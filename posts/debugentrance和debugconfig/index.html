йҰ–е…ҲжҳҜдёҖж®өжқҘиҮӘиҸңйёҹж•ҷзЁӢзҡ„жҸҸиҝ°пјҡ

еұһжҖ§е§”жүҳжҢҮзҡ„жҳҜдёҖдёӘзұ»зҡ„жҹҗдёӘеұһжҖ§еҖјдёҚжҳҜеңЁзұ»дёӯзӣҙжҺҘиҝӣиЎҢе®ҡд№үпјҢиҖҢжҳҜе°Ҷе…¶жүҳд»ҳз»ҷдёҖдёӘд»ЈзҗҶзұ»пјҢд»ҺиҖҢе®һзҺ°еҜ№иҜҘзұ»еұһжҖ§зҡ„з»ҹдёҖз®ЎзҗҶгҖӮ

val/var <еұһжҖ§еҗҚ>: <зұ»еһӢ> by <иЎЁиҫҫејҸ>

by е…ій”®еӯ—д№ӢеҗҺзҡ„иЎЁиҫҫејҸе°ұжҳҜе§”жүҳ, еұһжҖ§зҡ„ get() ж–№жі•(д»ҘеҸҠset() ж–№жі•)е°Ҷиў«е§”жүҳз»ҷиҝҷдёӘеҜ№иұЎзҡ„ getValue() е’Ң setValue() ж–№жі•гҖӮеұһжҖ§е§”жүҳдёҚеҝ…е®һзҺ°д»»дҪ•жҺҘеҸЈ, дҪҶеҝ…йЎ»жҸҗдҫӣ getValue() еҮҪж•°(еҜ№дәҺ varеұһжҖ§,иҝҳйңҖиҰҒ setValue() еҮҪж•°)гҖӮ

еҖҹеҠ©иҝҷдёӘзү№еҫҒпјҢжҲ‘д»¬еҸҜд»Ҙе®ҡд№үиҝҷж ·дёҖдёӘе§”жүҳзұ»пјҡ

class ZeroConfigDelegate<T>(private val clazz: Class<T>) {
    operator fun getValue(thisRef: Any?, property: KProperty<*>): T? =
        ZeroConfigHelper.readConfig(clazz)
    operator fun setValue(thisRef: Any?, property: KProperty<*>, value: T?) =
        ZeroConfigHelper.saveConfig(clazz, value)
}

иҝҷйҮҢд»Јз Ғзӣ®еүҚе®һзҺ°еҫҲз®ҖеҚ•пјҢе°ұжҳҜзӣҙжҺҘиҪ¬жҚўжҲҗjsonз„¶еҗҺеӯҳеҲ°spйҮҢгҖӮ

fun <T> saveConfig(clazz: Class<*>, value: T) {
    bufferMap[clazz] = value
    sp.edit {
        putString(getKeyOfClass(clazz), gson.toJson(value))
    }
}
fun <T> readConfig(clazz: Class<*>): T {
    return if (bufferMap.containsKey(clazz)) {
        bufferMap[clazz]
    } else {
        val jsonString = sp.getString(getKeyOfClass(clazz), "{}")
        gson.fromJson(jsonString, clazz)
    } as T
}

жіЁи§Је®ҡд№ү

жіЁж„ҸдёҠйқўжңүдёҖдёӘgetKeyOfClass(clazz)пјҢиҝҷдёӘж–№жі•жҳҜжҖҺд№Ҳе®һзҺ°зҡ„пјҹ

е…¶е®һеҰӮжһңз…§зӣ®еүҚдёәжӯўпјҢеҸӘиҰҒдҝқиҜҒе®ҡд№үзҡ„й…ҚзҪ®еӯ—ж®өkeyдә’дёҚзӣёеҗҢе°ұиЎҢдәҶпјҢйӮЈд№ҲеҸҜд»ҘзӣҙжҺҘз”Ёclazz.canonicalNameгҖӮдёҚиҝҮжҲ‘д»¬иҝҷйҮҢзҡ„йңҖжұӮиҝҳеёҢжңӣе®һзҺ°дёҖдёӘиғҪзӣҙжҺҘеңЁжүӢжңәж“ҚдҪңзҡ„з®ЎзҗҶз•ҢйқўпјҢжүҖд»Ҙз”ЁжіЁи§ЈеҺ»е®ҡд№үдёӢй…ҚзҪ®еӯ—ж®өзӣёе…ізҡ„дҝЎжҒҜдјҡжҜ”иҫғеҘҪгҖӮ

йҰ–е…Ҳж–°е»әдёҖдёӘkotlinжЁЎеқ—пјҲжіЁж„ҸдёҚжҳҜе®үеҚ“жЁЎеқ—пјүпјҢеҸ«еҒҡlibzeroconfigпјҢз”ЁжқҘж”ҫжҲ‘д»¬зҡ„жіЁи§ЈпјҢиҝҷж ·еҗҺйқўз”ЁеҲ°зҡ„ең°ж–№зӣҙжҺҘеҜје…ҘиҝҷдёӘжЁЎеқ—е°ұиЎҢдәҶгҖӮ

еҸӮиҖғеӯ—иҠӮзҡ„ABManagerпјҢжҲ‘иҝҷж ·е®ҡд№үжҲ‘зҡ„жіЁи§Ј

package top.ntutn.libzeroconfig
import kotlin.reflect.KClass
/**
 * ж ҮжіЁдәҺй…ҚзҪ®е®һдҪ“зұ»д№ӢдёҠпјҢжҢҮе®ҡй…ҚзҪ®еӯ—ж®өеҗҚ
 * @param key й…ҚзҪ®еӯ—ж®ө
 * @param title й…ҚзҪ®йЎ№еҗҚпјҲз»ҷдәәзңӢзҡ„пјү
 * @param owner иҙҹиҙЈдәә
 * @param scope жүҖеұһзҡ„дёҡеҠЎзәҝ
 */
@Target(AnnotationTarget.CLASS)
@Retention(AnnotationRetention.SOURCE)
annotation class ZeroConfig(
    val key: String,
    val title: String = "",
    val owner: String,
    val scope: KClass<out ZeroScope> = DefaultScope::class
)

иҰҒжұӮkeyе’Ңownerеҝ…йЎ»еЎ«еҶҷпјҢtitleжҳҜиҝҷдёӘй…ҚзҪ®еңЁжң¬ең°з®ЎзҗҶйқўжқҝдёӯжҳҫзӨәзҡ„еҗҚеӯ—гҖӮдёҖзӣ®дәҶз„¶гҖӮ

иҝҷдёӘжіЁи§ЈиҰҒж ҮеңЁе®ҡд№үзҡ„й…ҚзҪ®е®һдҪ“зұ»дёҠпјҢжіЁж„ҸеҰӮжһңжҳҜdata classиҰҒжұӮжүҖжңүеӯ—ж®өйғҪжҸҗдҫӣй»ҳи®ӨеҖјпјҢеӣ дёәеҗҰеҲҷзҡ„иҜқgsonеҸҚе°„еҲӣе»әеҜ№иұЎзҡ„ж—¶еҖҷдјҡеӣ дёәdata classжІЎжңүй»ҳи®Өжһ„йҖ еҮҪж•°йҒҮеҲ°й—®йўҳгҖӮиҷҪз„¶иҝҷдёӘй—®йўҳеҸҜд»ҘйҖҡиҝҮеә”з”Ёkotlin-noargжҸ’д»¶и§ЈеҶіпјҢдҪҶжҲ‘и§үеҫ—ејәеҲ¶иҰҒжұӮжүҖжңүй…ҚзҪ®зұ»йғҪжҸҗдҫӣжүҖжңүеӯ—ж®өзҡ„й»ҳи®ӨеҖјд№ҹдёҚй”ҷгҖӮ

жіЁи§Јзҡ„зј–иҜ‘жңҹеӨ„зҗҶ

RetentionжҢҮе®ҡзҡ„еҗҲйҖӮпјҢжіЁи§ЈжҳҜеҸҜд»Ҙиў«еёҰеҲ°иҝҗиЎҢжңҹй—ҙзҡ„гҖӮзңӢspringbootзҡ„дёҖеӨ§зҘЁжіЁи§ЈпјҢдёҠйқўжҢҮе®ҡзҡ„йғҪжҳҜ@Retention(RetentionPolicy.RUNTIME)гҖӮ

дҪҶдҪ еҸҜд»ҘжіЁж„ҸеҲ°жҲ‘дёҠйқўжіЁи§Је®ҡд№үзҡ„д»Јз Ғе№¶жІЎжңүиҝҷд№ҲеҒҡпјҢе®үеҚ“дёҠеӨ§е®¶еҹәжң¬йғҪе°ҪйҮҸдёҚиҝҷд№ҲеҒҡгҖӮspringbootеҸҜд»ҘеңЁиҝҗиЎҢж—¶йҖ’еҪ’жү«жҸҸжіЁи§ЈпјҢдҪҶжҲ‘д»¬е®үеҚ“жүӢжңәдёҠжқҘиҜҙиҝҷдёӘејҖй”Җе°ұеӨӘеӨ§дәҶгҖӮ

жүҖд»ҘиҰҒеңЁзј–иҜ‘жңҹй—ҙжҠҠиҝҷдёӘдәӢжғ…пјҲжү«жҸҸжіЁи§ЈдҝЎжҒҜпјүеҠһеҰҘпјҢжүҖд»Ҙе°ұиҰҒз”ЁеҲ°kaptдәҶпјҲе®үеҚ“дёҠжӯӨеүҚиҝҷдёӘдәӢжғ…жҳҜз”ЁannotationProcessorеӨ„зҗҶпјү

kapt еҚі Kotlin annotation processing toolпјҲKotlin жіЁи§ЈеӨ„зҗҶе·Ҙе…·пјүзј©еҶҷгҖӮйҖҡиҝҮе®ҡд№үжіЁи§ЈеӨ„зҗҶеҷЁеҸҜд»ҘеңЁзј–иҜ‘ж—¶еҜ№жәҗд»Јз ҒиҝӣиЎҢжЈҖжөӢз”ҹжҲҗйўқеӨ–зҡ„жәҗж–Үд»¶е’Ңе…¶д»–ж–Үд»¶пјҢд№ӢеҗҺзј–иҜ‘з”ҹжҲҗзҡ„жәҗж–Үд»¶е’ҢеҺҹжқҘзҡ„жәҗж–Үд»¶дёҖиө·з”ҹжҲҗclassж–Үд»¶гҖӮ еҗ¬дёҠеҺ»еҫҲcoolзҡ„ж“ҚдҪңпјҢдҪҝдәәдёҚз”ұиҮӘдё»иҒ”жғіеҲ°еҰӮжһңз”ҹжҲҗзҡ„жәҗж–Үд»¶иҝҳжңүиҝҷдёӘжіЁи§Је’ӢеҠһвҖҰвҖҰзӯ”жЎҲжҳҜkaptдјҡжү§иЎҢеӨҡж¬ЎпјҢзӣҙеҲ°жІЎжңүж–°зҡ„жіЁи§ЈеҸ‘зҺ°дёәжӯўгҖӮиҷҪз„¶иҝҷеҜ№жҲ‘д»¬иҝҷдёӘйңҖжұӮжІЎе•Ҙз”ЁеӨ„гҖӮ

д»Қз„¶жҳҜе®ҡд№үдёҖдёӘkotlinжЁЎеқ—пјҢиҝҷж¬ЎеҸ«libzeroconfigcompilerеҗ§гҖӮеңЁиҝҷйҮҢжҲ‘д»¬е®ҡд№үжҲ‘д»¬зҡ„жіЁи§ЈеӨ„зҗҶеҷЁгҖӮ

е…¶е®һиҰҒеҒҡзҡ„дәӢжғ…е°ұдёӨд»¶пјҡ

  1. е®ҡд№үдёҖдёӘзұ»з»§жүҝиҮӘAbstractProcessor
  2. жҠҠдҪ зҡ„зұ»еҗҚпјҲеёҰеҢ…еҗҚпјүеҶҷеҲ°META-INF/service/javax.annotation.processing.Processorдёӯ

еҜ№дәҺз¬¬дәҢжӯҘпјҢGoogleжҸҗдҫӣдәҶдёҖдёӘеҸ«auto serviceзҡ„дёңиҘҝеҸҜд»Ҙеё®жҲ‘д»¬з”ҹжҲҗиҝҷдёӘж–Үд»¶пјҢеҸӘиҰҒеј•е…ҘеҗҺеңЁдҪ зҡ„Annotation Processorзұ»дёҠеҠ дёҠ@AutoService(Processor.class)е°ұеҸҜд»ҘдәҶгҖӮ

иҮідәҺиҝҷдёӘжіЁи§ЈеӨ„зҗҶзұ»зҡ„е®һзҺ°пјҢе»әи®®иҝҳжҳҜеҲ«зңӢжҲ‘д»Јз ҒдәҶпјҢжңүдёӨдёӘжҲ‘и®ӨдёәеҫҲеҖјеҫ—еҸӮиҖғзҡ„йЎ№зӣ®пјҡдёҖдёӘжҳҜеүҚйқўжҸҗеҲ°зҡ„Auto ServiceпјҢеҸҰдёҖдёӘжҳҜEventBusзҡ„жіЁи§ЈеӨ„зҗҶеҷЁгҖӮ

override fun process(
    annotations: MutableSet<out TypeElement>,
    roundEnvironment: RoundEnvironment
): Boolean {
    counter++
    note("Processing round $counter, new annotations: ${annotations.isNotEmpty()}, processingOver: ${roundEnvironment.processingOver()}")
    if (roundEnvironment.processingOver() && annotations.isNotEmpty()) {
        error("Unexpected processing state: annotations still available after processing over")
        return false
    }
    if (annotations.isEmpty()) {
        return false
    }
    if (wasWrittenToFile) {
        error("Unexpected processing state: annotations still available after writing.")
        return false
    }
    // ж”¶йӣҶж•°жҚ®
    roundEnvironment.getElementsAnnotatedWith(ZeroConfig::class.java).forEach { element ->
        //дҪҝз”ЁдәҶжіЁи§Јзҡ„жҹҗдёӘзұ»
        if (element !is TypeElement) {
            error("жіЁи§ЈеҸӘиғҪж Үи®°еңЁе®һдҪ“зұ»дёҠпјҡ$element")
            return false
        }
        val annotation = element.getAnnotation(ZeroConfig::class.java)
        if (!checkAnnotationValid(annotation)) return false
        classInfoMap[annotation.key] = ZeroConfigInformation(
            key = annotation.key,
            clazz = element.qualifiedName.toString(),
            title = annotation.title,
            scope = getClassFromAnnotation { annotation.scope.qualifiedName!! },
            owner = annotation.owner
        )
    }
    generateCode()
    wasWrittenToFile = true
    return true
}

private fun getClassFromAnnotation(block: () -> String): String {
    return try {
        block()
    } catch (e: MirroredTypeException) {
        e.typeMirror.toString()
    }
}
